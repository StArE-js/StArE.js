<script type="text/javascript">

	function resetTabs() {
		$("#content > div").hide();
		$("#tabs a").attr("id", "");
		HideHeaders();
	}
	var myUrl = window.location.href;
	var showAll = false;
	var picTitle;
	var myUrlTab = myUrl.substring(myUrl.indexOf("#"));
	var myUrlTabName = myUrlTab.substring(0, 4);
	(function() {
		$("#content > div").hide();
		$("#tabs li:first a").attr("id", "current");
		$("#content > div:first").fadeIn();
		$("#tabs a").on("click", function(e) {
			e.preventDefault();
			if ($(this).attr("id") == "current") {
				return
			} else {
				resetTabs();
				$(this).attr("id", "current");
				$($(this).attr("name")).fadeIn()
			}
		});
		$("#mainPic a").on("click", function(e) {
			if(!showAll){
				myUrlTab = "#tabImages";
				resetTabs();
				$("a[name='" + myUrlTab + "']").attr("id", "current");
				$(myUrlTab).fadeIn();
			}
		});
		for (i = 1; i <= $("#tabs li").length; i++) {
			if (myUrlTab == myUrlTabName + i) {
				resetTabs();
				$("a[name='" + myUrlTab + "']").attr("id", "current");
				$(myUrlTab).fadeIn()
			}
		}
	})();
	
	function HideHeaders(){		
		$(".subHeading").hide();
	}
	
	function ShowAll(){		
		showAll = true;
		$("#tabs").find("li").each(function(){
			var name = $(this).children("a").attr("name");
			var tab = $("#content").find(name);
			tab.find(".subHeading").show();
			tab.fadeIn();
		});
		
		$("#mainPic a").bind('click', false);
		picTitle = $("#mainPic a").attr("title");
		$("#mainPic a").attr("title", "");
		$("#mainPic a").removeAttr("href");
		
		$("#tabs").hide();
		$("#ShowAllButton").hide();
		$("#ShowTabsButton").fadeIn();
 	}
	
	function ShowTabs(){
		showAll = false;
		$("#tabs").find("li").each(function(){
			var name = $(this).children("a").attr("name");
			var tab = $("#content").find(name);
			tab.find(".subHeading").hide();
			tab.hide();
		});
		
		$("#mainPic a").bind('click', true);		
		$("#mainPic a").attr("title", picTitle);
		$("#mainPic a").attr("href", "#");
		
		$("#tabs").fadeIn();
		$("#ShowAllButton").fadeIn();
		$("#ShowTabsButton").hide();
		resetTabs();
		$("#content > div:first").fadeIn();
	}
</script>
