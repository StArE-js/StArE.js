	function getCookieinGTM(cname) {
		var name = cname + "=";
		var ca = document.cookie.split(';');
		for(var i=0; i<ca.length; i++) {
				var c = ca[i];
				while (c.charAt(0)==' ') c = c.substring(1);
				if (c.indexOf(name) == 0) return c.substring(name.length, c.length);
		}
		return "";
	}
	
	/**
	* Set source tracking cookies
	*/
	function setCookieSourceTracking(name, value,days){
		var today = new Date();
		today.setTime(today.getTime() + (days*24*60*60*1000)); //90 days expiry time 
		var expires = "expires="+today.toGMTString();
		document.cookie = name + "=" + decodeURIComponent(value)+";"+expires+";domain=.pardeehomes.com;path=/;";
		
	}

	function getURLParamsForSourceTracking(sPageURL, name){
		var vars = [], hash;
		var hashes = sPageURL.slice(sPageURL.indexOf('?') + 1).split('&');
		for(var i = 0; i < hashes.length; i++){
			hash = hashes[i].split('=');
			if(hash[0].toLowerCase() == name || hash[0] == name){
				return hash[1];
			}
		}
		return "";
	}
